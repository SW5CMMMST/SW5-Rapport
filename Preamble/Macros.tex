\newcommand{\filefigure}[3]{
    \begin{figure}[H]
        \centering \footnotesize
        \input{#1}
        \caption{#2}
        \label{fig:#3}
    \end{figure}
}

\newcommand{\figur}[4]{
    \begin{figure}[H]
        \centering
        \includegraphics[width=#2\textwidth]{Figures/#1}
        \caption{#3}
        \label{fig:#4}
    \end{figure}
}

    \end{wrapfigure}
}
