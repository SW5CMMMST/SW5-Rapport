This Appendix contains the code used for the UPPAAL CCUC model.

caption={Local code for Device in CCUC.}, ]
// Place local declarations here.
clock x;
clock transmit_time;
int k = -1;                     //Timeslot
bool Connected = false;
int join_Count = 0;      
int Min_Units_For_Join = 0;
int WaitLoopCount = 0;
int backoff = 0;
int FailCount = 0;
int StartLoopCount = 0;
int StartAttempts = 0;
int FrameCount = 0;
int randomBackoff;
int local_network_id= -1;
int transmissionNumber = 1;

// Local copies of globals
int local_n = 0;                // Number of devices connected
int local_i = 0;               // Current time slot in the frame

void increment_Slot(){
    local_i = (local_i % local_n)+1;
}

void receive()
{
    local_i = i;
    local_n = n;
    local_network_id = network_id;
    x:=Delta_Proc+(Real_Tx_Time*global_transmissionNumber);
}

void attempt_Join_Network(){
    global_transmissionNumber = transmissionNumber;
    i = local_i;
    joining_ID = id;
}

void join_Network(){
    if(transmissionNumber == 1){
        k=n;
        n = n+1;
        i = local_i;
        local_n = local_n+1;
        Connected = true;
        joining_ID = -1;
        Connected_Counter++;
    }
    global_transmissionNumber = transmissionNumber;
}

void calc_Min_Validate(){
    if(local_n == 2) {
        Min_Units_For_Join = 1;
    } else {
        Min_Units_For_Join = (local_n-1)/2;
    }
}

void create_Network(){
    k = 1;
    i = 0;
    local_i = 0;
    local_n = 2;
    Connected = true;
    x:=0;
    Connected_Counter++;
    local_network_id = id;
}

void make_Payload(){
    global_transmissionNumber = transmissionNumber;
    i = local_i;
    n = local_n;
    network_id = local_network_id;
}

void ClearJoiningID(){
    if(joining_ID == id)
        joining_ID = -1;
}
